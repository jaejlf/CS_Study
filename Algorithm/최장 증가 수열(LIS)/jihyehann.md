using namespace std;

int n, arr[1001];
int lis() {
    int dp[1001] = {0};
    for(int i=1; i<n; i++) {
        for(int j=i-1; j>=0; j--) {
            if(arr[i] > arr[j]) 
            	dp[i] = max(dp[i], dp[j]+1);
        }
    }

    int answer = 0;
    for(int i=0; i<n; i++)
        answer = max(answer, dp[i]);
    
    return answer+1;
}
